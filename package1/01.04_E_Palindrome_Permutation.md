The idea is to count and calculate.
There should be only one odd count allowed.

``` Java
class Solution {
    public boolean canPermutePalindrome(String s) {
        char[] arr=s.toCharArray();
        int[] res=new int[256];
        for(int i=0;i<arr.length;i++){
            int index=arr[i];
            if(res[index]==0)
                res[index]++;
            else
                res[index]--;
        }
        int oddCount = 0;
        for(int i=0;i<res.length;i++) {
            if(res[i]==1){
                if(oddCount==1)
                    return false;
                oddCount+=1;
            }
        }

        return true;
    }
}
```
