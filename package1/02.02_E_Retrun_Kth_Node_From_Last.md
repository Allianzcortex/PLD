class Solution {
    public int kthToLast(ListNode head, int k) {
        // ListNode dummyHead=new ListNode(-1);
        // dummyHead.next = head;
        ListNode slow = head;
        ListNode fast = head;

        while(k-->0)
            fast = fast.next;
        while(fast!=null) {
            fast = fast.next;
            slow = slow.next;
        }

        return slow.val;

    }
}
