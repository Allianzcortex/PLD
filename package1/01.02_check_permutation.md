class Solution {
    public boolean CheckPermutation(String s1, String s2) {
        if(s1.length()!=s2.length())
            return false;
        int[] judge=new int[256];
        for(int i=0;i<s1.length();i++){
            judge[s1.charAt(i)]++;
            judge[s2.charAt(i)]--;
        }
        for(int i=0;i<judge.length;i++)
            if(judge[i]!=0)
                return false;
        return true;
    }
}
