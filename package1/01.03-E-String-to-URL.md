2. Solution2. build a new array

```Java
class Solution {
    public String replaceSpaces(String S, int length) {
        int emptyCount = 0;
        char[] arr= S.toCharArray();

        for(int i=0;i<length;i++)
            if(arr[i]==' ')
                emptyCount+=1;
        
        char[] res = new char[length+emptyCount*2];
        int index=0;
        for(int i=0;i<length;i++) {
            if(arr[i]==' '){
                res[index++]='%';
                res[index++]='2';
                res[index++]='0';
            } else 
                res[index++]=arr[i];
        }

        return String.valueOf(res);
    }
}
```
